import pandas as pd

import \
    seaborn as sns  # arzędzie dedykowane wizualizacji danych nieprzestrzennych, Wykresy seaborn są wykresami matplotlib
import scipy.cluster.hierarchy as shc  # importing scipy.cluster.hierarchy for dendrogram

import matplotlib.pyplot as plt

# SPRAWDZ SCIEZKE!! Uwaga ten skrypt przetwarza dane  GroceryStoreDataSet
df = pd.read_csv('dane_test.csv')
# wyświetlimy informacje o wczytanych danych
print('liczba rekordów we wczytamym pliku wynosi:', len(df))
print(df.head())
print(df.shape)

#wykres rozrzutu punktów
sns.relplot(data=df, x="x", y="y")
plt.show()
#wykres macierzowy - kazdy z kazdym
sns.pairplot(data=df[["x", "y"]])
plt.show()

#dendogram
dendrogram = shc.dendrogram(shc.linkage(df, method='ward'))  # finding the optimal number of clusters using dendrogram
plt.title('Dendrogram')  # title of the dendrogram
plt.xlabel('nr instancji (wiersza w naszych danych)')  # label of the x-axis
plt.ylabel('Euclidean distances')  # label of the y-axis
plt.show()  # show the dendrogram
